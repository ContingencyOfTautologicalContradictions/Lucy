cmake_minimum_required(VERSION 3.28.3)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)
set(CMAKE_SUPPRESS_REGENERATION true)
message(STATUS "Building for ${CMAKE_BUILD_TYPE}...")
project(Lucy)
if(${CMAKE_HOST_SYSTEM_NAME} STREQUAL "Windows")
    set(CMAKE_SYSTEM_VERSION 10.0.22621.2428)
    file(GLOB_RECURSE LUCY_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}/../Source/*.cpp")
    add_executable(Lucy WIN32 ${LUCY_SOURCES})
    target_link_libraries(Lucy PRIVATE d3d12 dxgi1_6)
    target_compile_definitions(Lucy PRIVATE lucy_windows=)
elseif(${CMAKE_HOST_SYSTEM_NAME} STREQUAL "Darwin")
    set(CMAKE_OSX_ARCHITECTURE arm64)
    file(GLOB_RECURSE ResourceFiles "${CMAKE_CURRENT_SOURCE_DIR}/Resources")
    foreach(ResourceFile "${ResourceFiles}")
        get_filename_component(FullResourceFolder "${CMAKE_CURRENT_SOURCE_DIR}/Resources" REALPATH)
        get_filename_component(FullResourceFile "${ResourceFile}" REALPATH)
        file(RELATIVE_PATH ResourcePath "${FullResourceFolder}" "${FullResourceFile}")
        get_filename_component(NewResourcePath "${ResourcePath}" DIRECTORY)
        set_property(SOURCE "${ResourceFile}" PROPERTY MACOSX_PACKAGE_LOCATION "Resources/${NewResourcePath}")
        source_group("Resources/${NewResourcePath}" FILES "${ResourceFile}")
    endforeach(ResourceFile)
    file(GLOB_RECURSE LUCY_OBJC_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}/../Source/*.mm")
    file(GLOB_RECURSE LUCY_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}/../Source/*.cpp")
    add_executable(Lucy MACOSX_BUNDLE ${ResourceFiles} ${LUCY_OBJC_SOURCES} ${LUCY_SOURCES})
    set_target_properties(Lucy PROPERTIES MACOSX_BUNDLE_INFO_PLIST "${CMAKE_CURRENT_SOURCE_DIR}/Template.plist" XCODE_ATTRIBUTE_PRODUCT_BUNDLE_IDENTIFIER com.iristech.lucy)
    target_link_libraries(Lucy PRIVATE "-framework Foundation -framework AppKit -framework QuartzCore -framework Metal -framework IOKit")
    target_compile_definitions(Lucy PRIVATE lucy_mac=)
elseif(${CMAKE_HOST_SYSTEM_NAME} STREQUAL "Linux")
    file(GLOB_RECURSE LUCY_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}/../Source/*.cpp")
    add_executable(Lucy ${LUCY_SOURCES})
    target_compile_definitions(Lucy PRIVATE VK_USE_PLATFORM_LINUX_KHR=1 lucy_linux=)
    target_link_libraries(Lucy PRIVATE vulkan wayland)
endif()
if(MSVC)
    target_compile_options(Lucy PRIVATE -WX -W4)
else()
    target_compile_options(Lucy PRIVATE -Werror -Wall -Wextra -Wpedantic)
endif()
if(${CMAKE_BUILD_TYPE} STREQUAL "Debug")
    target_compile_definitions(Lucy PRIVATE lucy_debug=)
    if(MSVC)
        target_compile_options(Lucy PRIVATE -DEBUG:FULL)
    else()
        target_compile_options(Lucy PRIVATE -g)
    endif()
elseif(${CMAKE_BUILD_TYPE} STREQUAL "Release")
    target_compile_options(Lucy PRIVATE -flto)
    target_link_options(Lucy PRIVATE -flto)
    if(MSVC)
        target_compile_options(Lucy PRIVATE -GR- -EHsc- -Oy- -Ox)
    else()
        target_compile_options(Lucy PRIVATE -fno-rtti -fno-exceptions -fomit-frame-pointer -O3)
    endif()
endif()
set_target_properties(Lucy PROPERTIES CXX_EXTENSIONS OFF)
target_compile_features(Lucy PRIVATE cxx_std_23)
target_compile_definitions(Lucy PRIVATE lucy_major=0 lucy_middle=0 lucy_minor=1)
